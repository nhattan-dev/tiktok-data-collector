CREATE TABLE public.tk_ad (
	ad_id int8 NOT NULL,
	ad_format varchar(255) NULL,
	ad_name varchar(255) NULL,
	ad_text varchar(255) NULL,
	adgroup_id int8 NULL,
	adgroup_name varchar(255) NULL,
	advertiser_id int8 NULL,
	app_name varchar(255) NULL,
	campaign_id int8 NULL,
	campaign_name varchar(255) NULL,
	opt_status varchar(255) NULL,
	status varchar(255) NULL,
	CONSTRAINT tk_ad_pkey PRIMARY KEY (ad_id)
);

CREATE TABLE public.tk_adgroup (
	adgroup_id int8 NOT NULL,
	adgroup_name varchar(255) NULL,
	advertiser_id int8 NULL,
	age varchar(255) NULL,
	campaign_id int8 NULL,
	campaign_name varchar(255) NULL,
	create_time varchar(255) NULL,
	gender varchar(255) NULL,
	languages varchar(255) NULL,
	"location" varchar(255) NULL,
	modify_time varchar(255) NULL,
	opt_status varchar(255) NULL,
	schedule_end_time date NULL,
	schedule_start_time date NULL,
	status varchar(255) NULL,
	CONSTRAINT tk_adgroup_pkey PRIMARY KEY (adgroup_id)
);

CREATE TABLE public.tk_campaign (
	campaign_id int8 NOT NULL,
	advertiser_id int8 NULL,
	bid_type varchar(255) NULL,
	budget float4 NULL,
	budget_mode varchar(255) NULL,
	budget_optimize_switch int8 NULL,
	campaign_name varchar(255) NULL,
	create_time varchar(255) NULL,
	modify_time varchar(255) NULL,
	objective varchar(255) NULL,
	objective_type varchar(255) NULL,
	opt_status varchar(255) NULL,
	optimize_goal varchar(255) NULL,
	status varchar(255) NULL,
	CONSTRAINT tk_campaign_pkey PRIMARY KEY (campaign_id)
);
